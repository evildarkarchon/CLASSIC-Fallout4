<task id="2">
Implement tiered threshold analysis with config file support

Add step to parse critcmp JSON and apply thresholds from `rust/benchmark-config.yaml`:

```yaml
- name: Install yq for YAML parsing
  if: github.event_name == 'pull_request' && steps.baseline-check.outputs.baseline_exists == 'true'
  run: |
    # Install yq (YAML processor) for reading benchmark-config.yaml
    sudo wget -qO /usr/local/bin/yq https://github.com/mikefarah/yq/releases/latest/download/yq_linux_amd64
    sudo chmod +x /usr/local/bin/yq

- name: Analyze thresholds
  if: github.event_name == 'pull_request' && steps.baseline-check.outputs.baseline_exists == 'true'
  id: analyze
  working-directory: rust
  shell: bash
  run: |
    # Read default thresholds from config file (created in 17-01 Task 4)
    CONFIG_FILE="benchmark-config.yaml"
    if [ -f "$CONFIG_FILE" ]; then
      WARN_THRESHOLD=$(yq '.defaults.warning_threshold' "$CONFIG_FILE")
      FAIL_THRESHOLD=$(yq '.defaults.failure_threshold' "$CONFIG_FILE")
      echo "Loaded thresholds from config: warn=${WARN_THRESHOLD}%, fail=${FAIL_THRESHOLD}%"
    else
      echo "::warning::Config file not found, using hardcoded defaults"
      WARN_THRESHOLD=5
      FAIL_THRESHOLD=10
    fi

    if [ ! -f comparison.json ]; then
      echo "status=skipped" >> $GITHUB_OUTPUT
      echo "summary=No comparison data available" >> $GITHUB_OUTPUT
      exit 0
    fi

    # Initialize counters and arrays
    has_failure=false
    has_warning=false
    warnings_md=""
    failures_md=""
    improvements_md=""

    # Function to get threshold for a benchmark (checks overrides)
    get_warn_threshold() {
      local name="$1"
      local override=$(yq ".overrides[\"$name\"].warning_threshold // \"\"" "$CONFIG_FILE" 2>/dev/null)
      if [ -n "$override" ] && [ "$override" != "null" ]; then
        echo "$override"
      else
        echo "$WARN_THRESHOLD"
      fi
    }

    get_fail_threshold() {
      local name="$1"
      local override=$(yq ".overrides[\"$name\"].failure_threshold // \"\"" "$CONFIG_FILE" 2>/dev/null)
      if [ -n "$override" ] && [ "$override" != "null" ]; then
        echo "$override"
      else
        echo "$FAIL_THRESHOLD"
      fi
    }

    # Parse each benchmark result
    # critcmp --export format: {"benchmarks": [{"name": "...", "baseline": {...}, "current": {...}}]}
    while IFS= read -r line; do
      name=$(echo "$line" | jq -r '.name // empty')
      if [ -z "$name" ]; then continue; fi

      # Get per-benchmark thresholds (with override support)
      bench_warn=$(get_warn_threshold "$name")
      bench_fail=$(get_fail_threshold "$name")

      # Get percentage change (positive = slower = regression)
      change=$(echo "$line" | jq -r '.comparison.mean.percentage_change // 0')
      current_time=$(echo "$line" | jq -r '.current.mean.estimate // "N/A"')
      baseline_time=$(echo "$line" | jq -r '.baseline.mean.estimate // "N/A"')

      # Format times for display (convert to human-readable)
      change_int=${change%.*}
      abs_change=${change_int#-}

      if [ "$change_int" -gt "$bench_fail" ]; then
        has_failure=true
        failures_md="${failures_md}| ${name} | +${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      elif [ "$change_int" -gt "$bench_warn" ]; then
        has_warning=true
        warnings_md="${warnings_md}| ${name} | +${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      elif [ "$change_int" -lt "-$bench_warn" ]; then
        improvements_md="${improvements_md}| ${name} | ${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      fi
    done < <(jq -c '.benchmarks[]' comparison.json 2>/dev/null || echo "")

    # Set outputs
    echo "has_failure=$has_failure" >> $GITHUB_OUTPUT
    echo "has_warning=$has_warning" >> $GITHUB_OUTPUT

    # Build status message
    if [ "$has_failure" = "true" ]; then
      echo "status=failure" >> $GITHUB_OUTPUT
    elif [ "$has_warning" = "true" ]; then
      echo "status=warning" >> $GITHUB_OUTPUT
    else
      echo "status=success" >> $GITHUB_OUTPUT
    fi

    # Save markdown tables for PR comment (used by 17-03)
    echo "$failures_md" > failures.md
    echo "$warnings_md" > warnings.md
    echo "$improvements_md" > improvements.md
```

**Key change from original:** Thresholds are now read from `rust/benchmark-config.yaml` using yq, with per-benchmark override support. This enables the user's locked decision for custom thresholds on noisy benchmarks.
</task>

<task id="3">
Add label bypass check

Add step to check for `perf-regression-accepted` label:

```yaml
- name: Check for bypass label
  if: github.event_name == 'pull_request'
  id: check-label
  shell: bash
  run: |
    labels='${{ toJSON(github.event.pull_request.labels.*.name) }}'
    if echo "$labels" | jq -e 'map(select(. == "perf-regression-accepted")) | length > 0' > /dev/null 2>&1; then
      echo "bypass=true" >> $GITHUB_OUTPUT
      echo "::notice::Regression accepted via perf-regression-accepted label"
    else
      echo "bypass=false" >> $GITHUB_OUTPUT
    fi
```

This bypass is checked by Plan 17-03 before failing the build.
</task>
