**Analyze thresholds (reads from benchmark-config.yaml):**
```yaml
- name: Analyze thresholds
  if: github.event_name == 'pull_request' && steps.baseline-check.outputs.baseline_exists == 'true'
  id: analyze
  working-directory: rust
  shell: bash
  run: |
    # Read default thresholds from config file
    CONFIG_FILE="benchmark-config.yaml"
    if [ -f "$CONFIG_FILE" ]; then
      WARN_THRESHOLD=$(yq '.defaults.warning_threshold' "$CONFIG_FILE")
      FAIL_THRESHOLD=$(yq '.defaults.failure_threshold' "$CONFIG_FILE")
      echo "Loaded thresholds from config: warn=${WARN_THRESHOLD}%, fail=${FAIL_THRESHOLD}%"
    else
      echo "::warning::Config file not found, using hardcoded defaults"
      WARN_THRESHOLD=5
      FAIL_THRESHOLD=10
    fi

    if [ ! -f comparison.json ]; then
      echo "status=skipped" >> $GITHUB_OUTPUT
      echo "summary=No comparison data available" >> $GITHUB_OUTPUT
      exit 0
    fi

    # Initialize counters
    has_failure=false
    has_warning=false
    warnings_md=""
    failures_md=""
    improvements_md=""

    # Functions to get thresholds (with override support)
    get_warn_threshold() {
      local name="$1"
      local override=$(yq ".overrides[\"$name\"].warning_threshold // \"\"" "$CONFIG_FILE" 2>/dev/null)
      if [ -n "$override" ] && [ "$override" != "null" ]; then
        echo "$override"
      else
        echo "$WARN_THRESHOLD"
      fi
    }

    get_fail_threshold() {
      local name="$1"
      local override=$(yq ".overrides[\"$name\"].failure_threshold // \"\"" "$CONFIG_FILE" 2>/dev/null)
      if [ -n "$override" ] && [ "$override" != "null" ]; then
        echo "$override"
      else
        echo "$FAIL_THRESHOLD"
      fi
    }

    # Parse each benchmark result
    while IFS= read -r line; do
      name=$(echo "$line" | jq -r '.name // empty')
      if [ -z "$name" ]; then continue; fi

      bench_warn=$(get_warn_threshold "$name")
      bench_fail=$(get_fail_threshold "$name")

      change=$(echo "$line" | jq -r '.comparison.mean.percentage_change // 0')
      current_time=$(echo "$line" | jq -r '.current.mean.estimate // "N/A"')
      baseline_time=$(echo "$line" | jq -r '.baseline.mean.estimate // "N/A"')

      change_int=${change%.*}
      abs_change=${change_int#-}

      if [ "$change_int" -gt "$bench_fail" ]; then
        has_failure=true
        failures_md="${failures_md}| ${name} | +${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      elif [ "$change_int" -gt "$bench_warn" ]; then
        has_warning=true
        warnings_md="${warnings_md}| ${name} | +${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      elif [ "$change_int" -lt "-$bench_warn" ]; then
        improvements_md="${improvements_md}| ${name} | ${change_int}% | ${current_time} | ${baseline_time} |"$'\n'
      fi
    done < <(jq -c '.benchmarks[]' comparison.json 2>/dev/null || echo "")

    # Set outputs
    echo "has_failure=$has_failure" >> $GITHUB_OUTPUT
    echo "has_warning=$has_warning" >> $GITHUB_OUTPUT

    if [ "$has_failure" = "true" ]; then
      echo "status=failure" >> $GITHUB_OUTPUT
    elif [ "$has_warning" = "true" ]; then
      echo "status=warning" >> $GITHUB_OUTPUT
    else
      echo "status=success" >> $GITHUB_OUTPUT
    fi

    # Save markdown tables for PR comment (used by 17-03)
    echo "$failures_md" > failures.md
    echo "$warnings_md" > warnings.md
    echo "$improvements_md" > improvements.md
```

Thresholds are read from `rust/benchmark-config.yaml` using yq, with per-benchmark override support.
