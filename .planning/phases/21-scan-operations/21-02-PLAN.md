**Enhance ScanResult for rich status formatting:**

1. Add format_status() method to ScanResult:
   ```rust
   impl ScanResult {
       /// Format status message for display
       /// - "No crash logs found" for empty discovery
       /// - "Scanned N logs" for success
       /// - "Scanned N logs (M errors)" for success with errors
       /// - "Cancelled (X of Y logs)" for cancellation
       pub fn format_status(&self) -> String {
           if self.total == 0 {
               "No crash logs found".to_string()
           } else if self.cancelled {
               format!("Cancelled ({} of {} logs)", self.attempted, self.total)
           } else if self.error_count > 0 {
               format!("Scanned {} logs ({} errors)", self.total, self.error_count)
           } else {
               format!("Scanned {} logs", self.total)
           }
       }

       /// Check if scan completed successfully with results
       pub fn has_results(&self) -> bool {
           !self.cancelled && !self.reports.is_empty()
       }
   }
   ```

**Update the scan completion handler in setup_scan_callbacks:**

1. Add auto-switch to Results tab (index 1) on successful completion with results:
   ```rust
   AsyncBridge::run_with_ui_update(
       classic_gui::scan_crash_logs(window_weak.clone(), cancel_token, crash_log_path),
       move |result| {
           if let Some(w) = window_weak.upgrade() {
               match result {
                   Ok(scan_result) => {
                       w.set_scan_progress(100.0);
                       w.set_scan_status(scan_result.format_status().into());
                       w.set_scan_in_progress(false);

                       // Auto-switch to Results tab ONLY on successful completion with results
                       // Per CONTEXT.md: "Zero logs: stay on Main Options (no auto-switch)"
                       if scan_result.has_results() {
                           w.set_active_tab_index(1); // Results tab
                       }

                       // TODO: Store scan_result.reports in AppState for Phase 22
                   }
                   Err(msg) => {
                       w.set_scan_progress(0.0);
                       w.set_scan_status(msg.into());
                       w.set_scan_in_progress(false);
                       // Error case: stay on Main Options, no auto-switch
                   }
               }

               // Auto-clear status bar after delay
               // Per CONTEXT.md: "Status bar auto-clears after delay when scan completes"
               let window_weak_for_clear = window_weak.clone();
               AsyncBridge::spawn_background(async move {
                   tokio::time::sleep(std::time::Duration::from_secs(5)).await;
                   let _ = window_weak_for_clear.upgrade_in_event_loop(|w| {
                       // Only clear if not currently scanning (user may have started another scan)
                       if !w.get_scan_in_progress() {
                           w.set_scan_status("Ready".into());
                       }
                   });
               });
           }
       },
   );
   ```
