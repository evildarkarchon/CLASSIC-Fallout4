```python
"""VAL-03: Rust report generation matches golden report format.

Tests that Rust OrchestratorCore produces AUTOSCAN.md output that matches
golden files character-for-character (after timestamp masking and path
normalization).

Note: Since Phase 9 deleted Python orchestrator, golden files capture Rust
output. These tests serve as regression tests ensuring Rust output remains
stable across code changes.

Per CONTEXT.md decisions:
- Timestamps: Masked with {{TIMESTAMP}} placeholder
- Paths: Normalized to forward slashes (not masked)
- Whitespace: Strict matching
- Comparison: Whole-file (entire report as one unit)
"""

import pytest
from pathlib import Path
from tests.fixtures.golden_fixtures import (
    GOLDEN_DIR,
    mask_dynamic_data,
    normalize_paths,
    generate_diff,
)
from ClassicLib.integration.rust.orchestrator_api import ClassicOrchestrator

# Golden report files (subset of logs that produce valid reports)
# Not all 16 logs may produce complete reports - filter to those that do
REPORT_GOLDEN_STEMS = []  # Populated after capture script runs


def get_report_golden_stems() -> list[str]:
    """Discover available report golden files."""
    if REPORT_GOLDEN_STEMS:
        return REPORT_GOLDEN_STEMS

    # Find all *_report.golden.md files
    for golden_file in GOLDEN_DIR.glob("*_report.golden.md"):
        stem = golden_file.stem.replace("_report.golden", "")
        REPORT_GOLDEN_STEMS.append(stem)

    return REPORT_GOLDEN_STEMS


def normalize_for_comparison(text: str) -> str:
    """Apply all normalization for parity comparison."""
    result = mask_dynamic_data(text)
    result = normalize_paths(result)
    return result


def golden_stem_to_log_name(stem: str) -> str:
    """Convert golden file stem to original log filename."""
    # Same mapping as test_scanning_parity.py
    special_cases = {
        "crash_123_1": "crash-$123 1.log",
        "crash_0akensh1eld_1": "crash-0akensh1eld 1.log",
        "crash_0DB9300": "crash-0DB9300.log",
        "crash_12624": "crash-12624.log",
        "crash_1D1777B": "crash-1D1777B.log",
        "crash_2022_10_07_01_50_25_Lancer_Vance": "crash-2022-10-07-01-50-25 Lancer_Vance.log",
        "crash_2022_10_10_02_12_34_Bones": "crash-2022-10-10-02-12-34 Bones.log",
        "crash_2022_10_16_22_55_26_Arcanist": "crash-2022-10-16-22-55-26 Arcanist.log",
    }
    if stem in special_cases:
        return special_cases[stem]
    return stem.replace("_", "-") + ".log"


@pytest.fixture
def orchestrator():
    """Create ClassicOrchestrator for report generation."""
    return ClassicOrchestrator()


@pytest.fixture
def sample_logs_dir() -> Path:
    """Path to sample crash logs directory."""
    return Path(__file__).parent.parent.parent / "sample_logs" / "FO4"


@pytest.mark.parity
@pytest.mark.integration
class TestReportParity:
    """VAL-03: Rust report generation matches golden report format."""

    @pytest.mark.parametrize("log_stem", get_report_golden_stems())
    def test_full_report_parity(
        self, log_stem: str, orchestrator, sample_logs_dir: Path
    ):
        """Rust report output matches golden report for {log_stem}."""
        # Load golden report
        golden_path = GOLDEN_DIR / f"{log_stem}_report.golden.md"
        if not golden_path.exists():
            pytest.skip(f"Golden report not found: {golden_path}")

        expected_report = golden_path.read_text(encoding="utf-8")

        # Find corresponding log file
        log_name = golden_stem_to_log_name(log_stem)
        log_path = sample_logs_dir / log_name

        if not log_path.exists():
            pytest.skip(f"Sample log not found: {log_path}")

        # Generate report with Rust orchestrator
        result = orchestrator.process_crash_log(log_path)
        actual_report = "\\n".join(result.report_lines)

        # Normalize both for comparison
        actual_normalized = normalize_for_comparison(actual_report)
        expected_normalized = normalize_for_comparison(expected_report)

        if actual_normalized != expected_normalized:
            diff = generate_diff(expected_normalized, actual_normalized)
            pytest.fail(
                f"Report parity mismatch for {log_stem}:\\n\\n{diff}\\n\\n"
                "Golden files are authoritative - Rust must match them."
            )

    def test_orchestrator_is_feature_complete(self, orchestrator):
        """VAL-05: Orchestrator has full feature parity."""
        assert orchestrator.is_feature_complete(), (
            "Rust orchestrator is not feature complete. "
            "Required: plugin analyzer + suspect scanner available."
        )
```

```python
        """PathValidator rejects nonexistent paths."""

        result = validator.validate_path(r"Z:\\NonExistent\\Path")
```
