fn main() {
    let config = slint_build::CompilerConfiguration::new()
        .with_style("fluent-dark".into());

    slint_build::compile_with_config("ui/main.slint", config)

}

Create a reusable PathInput component in a new widgets/ directory:

```slint
// PathInput - Reusable path input with browse button
import { Button, LineEdit, HorizontalBox } from "std-widgets.slint";

export component PathInput inherits HorizontalBox {
    in-out property <string> path;
    in-out property <string> placeholder: "Select folder...";
    in-out property <bool> enabled: true;
    callback browse-clicked();

    spacing: 8px;

    LineEdit {
        text <=> root.path;
        placeholder-text: root.placeholder;
        enabled: root.enabled;
        horizontal-stretch: 1;
    }

    Button {
        text: "Browse...";
        enabled: root.enabled;
        clicked => { root.browse-clicked(); }
    }
}
```

This provides a consistent path input pattern used throughout the application.
