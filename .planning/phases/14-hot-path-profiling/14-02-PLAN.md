use std::sync::atomic::{AtomicU64, Ordering};
use serde::Serialize;

// Global counters for cache statistics
static CACHE_HITS: AtomicU64 = AtomicU64::new(0);
static CACHE_MISSES: AtomicU64 = AtomicU64::new(0);

/// Cache performance statistics.
#[derive(Debug, Clone, Serialize)]
pub struct CacheStats {
    pub hits: u64,
    pub misses: u64,
    pub hit_rate: f64,
    pub size: usize,
}

/// Get current cache statistics.
pub fn cache_stats() -> CacheStats {
    let hits = CACHE_HITS.load(Ordering::Relaxed);
    let misses = CACHE_MISSES.load(Ordering::Relaxed);
    let total = hits + misses;
    CacheStats {
        hits,
        misses,
        hit_rate: if total > 0 { hits as f64 / total as f64 } else { 0.0 },
        size: CACHE.len(),
    }
}

/// Reset cache statistics (for testing).
pub fn reset_cache_stats() {
    CACHE_HITS.store(0, Ordering::Relaxed);
    CACHE_MISSES.store(0, Ordering::Relaxed);
}

**run_dhat.ps1:**

Create PowerShell script for dhat heap profiling:

```powershell
<#
.SYNOPSIS
    Runs dhat heap profiling for Rust code.

.DESCRIPTION
    Builds and runs a Rust binary/test with dhat heap profiling enabled.
    Outputs heap allocation data to target/profiling/dhat/ directory.

.PARAMETER Crate
    Crate to profile (e.g., classic-yaml-core).

.PARAMETER Test
    Run tests instead of binary.

.PARAMETER TestFilter
    Filter tests by name when using -Test.

.PARAMETER Bench
    Run benchmarks (with dhat feature enabled).

.PARAMETER Output
    Custom output directory. Default: target/profiling/dhat/

.EXAMPLE
    .\run_dhat.ps1 -Crate classic-yaml-core -Test
    # Profile yaml-core tests with dhat

.EXAMPLE
    .\run_dhat.ps1 -Crate classic-settings-core -Test -TestFilter "test_load"
    # Profile specific test
#>

[CmdletBinding()]
param(
    [Parameter(Mandatory = $true)]
    [string]$Crate,

    [switch]$Test,

    [string]$TestFilter,

    [switch]$Bench,

    [string]$Output
)

Write-Host "========================================" -ForegroundColor Cyan
Write-Host " CLASSIC dhat Heap Profiler" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Setup output directory
$timestamp = Get-Date -Format "yyyy-MM-dd-HHmmss"
$outputDir = if ($Output) { $Output } else { "target/profiling/dhat" }
New-Item -ItemType Directory -Force -Path $outputDir | Out-Null

Write-Host "[Config] Crate: $Crate" -ForegroundColor Yellow
Write-Host "[Config] Output dir: $outputDir" -ForegroundColor Yellow

# Set DHAT output file via environment variable
$dhatFile = "$outputDir/dhat-heap-$Crate-$timestamp.json"
$env:DHAT_OUTPUT_FILE = $dhatFile
Write-Host "[Config] DHAT output: $dhatFile" -ForegroundColor Yellow

# Build cargo command
$cargoArgs = @()

if ($Test) {
    $cargoArgs += @("test", "-p", $Crate, "--features", "dhat-heap", "--release")
    Write-Host "[Config] Running tests" -ForegroundColor Yellow

    if ($TestFilter) {
        $cargoArgs += @("--", $TestFilter)
        Write-Host "[Config] Test filter: $TestFilter" -ForegroundColor Yellow
    }
} elseif ($Bench) {
    $cargoArgs += @("bench", "-p", $Crate, "--features", "dhat-heap")
    Write-Host "[Config] Running benchmarks" -ForegroundColor Yellow
} else {
    $cargoArgs += @("run", "-p", $Crate, "--features", "dhat-heap", "--release")
    Write-Host "[Config] Running binary" -ForegroundColor Yellow
}

Write-Host ""
Write-Host "[Command] cargo $($cargoArgs -join ' ')" -ForegroundColor DarkGray
Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

# Run with dhat
Push-Location rust
$startTime = Get-Date
try {
    & cargo @cargoArgs
    $exitCode = $LASTEXITCODE
} finally {
    Pop-Location
}
$endTime = Get-Date
$elapsed = $endTime - $startTime

Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

if ($exitCode -eq 0) {
    Write-Host "[Done] dhat profiling complete" -ForegroundColor Green
    Write-Host "[Duration] $($elapsed.ToString('hh\:mm\:ss\.fff'))" -ForegroundColor DarkGray

    # Check if dhat file was created
    if (Test-Path $dhatFile) {
        $fileSize = (Get-Item $dhatFile).Length
        Write-Host "[Output] $dhatFile ($fileSize bytes)" -ForegroundColor Green
        Write-Host ""
        Write-Host "View results at: https://nnethercote.github.io/dh_view/dh_view.html" -ForegroundColor Cyan
    } else {
        Write-Host "[Warning] DHAT output file not found. The crate may not have dhat enabled." -ForegroundColor Yellow
        Write-Host "Ensure the crate has:" -ForegroundColor DarkGray
        Write-Host '  [features] dhat-heap = ["dep:dhat"]' -ForegroundColor DarkGray
        Write-Host '  #[cfg(feature = "dhat-heap")] #[global_allocator] static ALLOC: dhat::Alloc = dhat::Alloc;' -ForegroundColor DarkGray
    }
} else {
    Write-Host "[Error] dhat profiling failed with exit code $exitCode" -ForegroundColor Red
    exit 1
}

Write-Host ""
```

**dump_cache_stats.ps1:**

Create PowerShell script to extract cache statistics via Python:

```powershell
<#
.SYNOPSIS
    Dumps cache statistics from CLASSIC Rust caches.

.DESCRIPTION
    Imports CLASSIC Python modules and extracts cache statistics
    from instrumented DashMap caches. Outputs to console and optionally JSON file.

.PARAMETER Output
    Optional JSON output file path.

.PARAMETER Format
    Output format: 'console' (default), 'json', 'both'.

.EXAMPLE
    .\dump_cache_stats.ps1
    # Console output of all cache stats

.EXAMPLE
    .\dump_cache_stats.ps1 -Format both -Output cache-stats.json
    # Console output and JSON file
#>

[CmdletBinding()]
param(
    [string]$Output,

    [ValidateSet('console', 'json', 'both')]
    [string]$Format = 'console'
)

Write-Host "========================================" -ForegroundColor Cyan
Write-Host " CLASSIC Cache Statistics" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Python script to extract cache stats
$pythonScript = @'
import json
import sys
from datetime import datetime

stats = {
    "timestamp": datetime.now().strftime("%Y-%m-%d-%H%M%S"),
    "caches": {}
}

# Try to get yaml cache stats
try:
    import classic_yaml
    ops = classic_yaml.RustYamlOperations()
    yaml_stats = ops.get_cache_stats()
    stats["caches"]["yaml"] = yaml_stats
except Exception as e:
    stats["caches"]["yaml"] = {"error": str(e)}

# Try to get settings cache stats
try:
    import classic_settings
    settings_stats = classic_settings.cache_stats()
    stats["caches"]["settings"] = {
        "hits": settings_stats.hits,
        "misses": settings_stats.misses,
        "hit_rate": settings_stats.hit_rate,
        "size": settings_stats.size,
    }
except Exception as e:
    stats["caches"]["settings"] = {"error": str(e)}

# Output format based on argument
output_format = sys.argv[1] if len(sys.argv) > 1 else "console"
output_path = sys.argv[2] if len(sys.argv) > 2 else None

if output_format in ("console", "both"):
    print("\n=== Cache Statistics ===\n")
    for cache_name, cache_stats in stats["caches"].items():
        print(f"[{cache_name}]")
        if "error" in cache_stats:
            print(f"  Error: {cache_stats['error']}")
        else:
            for key, value in cache_stats.items():
                if key == "hit_rate":
                    print(f"  {key}: {value:.2%}")
                else:
                    print(f"  {key}: {value}")
        print()

if output_format in ("json", "both") and output_path:
    with open(output_path, "w") as f:
        json.dump(stats, f, indent=2)
    print(f"JSON written to: {output_path}")
elif output_format == "json":
    print(json.dumps(stats, indent=2))
'@

# Setup output if specified
$outputArg = ""
if ($Output) {
    $outputDir = Split-Path $Output -Parent
    if ($outputDir -and -not (Test-Path $outputDir)) {
        New-Item -ItemType Directory -Force -Path $outputDir | Out-Null
    }
    $outputArg = $Output
} elseif ($Format -in @('json', 'both')) {
    $timestamp = Get-Date -Format "yyyy-MM-dd-HHmmss"
    $outputDir = "target/profiling/cache-stats"
    New-Item -ItemType Directory -Force -Path $outputDir | Out-Null
    $outputArg = "$outputDir/cache-stats-$timestamp.json"
}

# Run Python script
try {
    $pythonScript | uv run python - $Format $outputArg
    $exitCode = $LASTEXITCODE
} catch {
    Write-Host "[Error] $_" -ForegroundColor Red
    $exitCode = 1
}

if ($exitCode -ne 0) {
    Write-Host "[Error] Failed to get cache statistics" -ForegroundColor Red
    exit 1
}

Write-Host ""
```
