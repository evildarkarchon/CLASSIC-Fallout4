Create scripts/profile/run_flamegraph.ps1 following the pattern from run_benchmarks.ps1:

```powershell
<#
.SYNOPSIS
    Generates flamegraphs for Rust code profiling.

.DESCRIPTION
    Runs cargo flamegraph with configurable modes and output management.

    Modes:
    - quick (default): Lower sampling rate (99 Hz), shorter duration (10s)
    - thorough: Higher sampling rate (997 Hz), longer duration (60s)

.PARAMETER Mode
    Profiling mode: 'quick' (default) or 'thorough'.

.PARAMETER Crate
    Specific crate to profile (e.g., classic-yaml-core).

.PARAMETER Bench
    Profile a benchmark instead of binary.

.PARAMETER BenchFilter
    Filter benchmarks by name when using -Bench.

.PARAMETER Output
    Custom output path. Default: target/profiling/flamegraphs/flamegraph-{timestamp}.svg

.PARAMETER Open
    Open the generated SVG in default browser after generation.

.EXAMPLE
    .\run_flamegraph.ps1
    # Quick flamegraph of default binary

.EXAMPLE
    .\run_flamegraph.ps1 -Mode thorough -Bench -Crate classic-yaml-core
    # Thorough flamegraph of yaml benchmarks
#>

[CmdletBinding()]
param(
    [ValidateSet('quick', 'thorough')]
    [string]$Mode = 'quick',

    [string]$Crate,

    [switch]$Bench,

    [string]$BenchFilter,

    [string]$Output,

    [switch]$Open
)

# Set profile mode env var for consistency with benchmarks
$env:PROFILE_MODE = $Mode

Write-Host "========================================" -ForegroundColor Cyan
Write-Host " CLASSIC Flamegraph Profiler" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Determine sampling parameters based on mode
$rate = if ($Mode -eq 'thorough') { 997 } else { 99 }
$duration = if ($Mode -eq 'thorough') { 60 } else { 10 }

Write-Host "[Config] Mode: $Mode" -ForegroundColor Yellow
Write-Host "         Sampling rate: $rate Hz" -ForegroundColor DarkGray
Write-Host "         Duration: ${duration}s" -ForegroundColor DarkGray

# Setup output directory and file
$timestamp = Get-Date -Format "yyyy-MM-dd-HHmmss"
$outputDir = "target/profiling/flamegraphs"
New-Item -ItemType Directory -Force -Path $outputDir | Out-Null

$outputFile = if ($Output) {
    $Output
} else {
    "$outputDir/flamegraph-$timestamp.svg"
}

Write-Host "[Output] $outputFile" -ForegroundColor Yellow

# Build cargo flamegraph command
$flameArgs = @("flamegraph", "-o", $outputFile, "--freq", $rate)

if ($Crate) {
    $flameArgs += @("-p", $Crate)
    Write-Host "[Config] Crate: $Crate" -ForegroundColor Yellow
}

if ($Bench) {
    $flameArgs += "--bench"
    Write-Host "[Config] Profiling benchmark" -ForegroundColor Yellow

    if ($BenchFilter) {
        $flameArgs += $BenchFilter
        Write-Host "[Config] Bench filter: $BenchFilter" -ForegroundColor Yellow
    }
}

Write-Host ""
Write-Host "[Command] cargo $($flameArgs -join ' ')" -ForegroundColor DarkGray
Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

# Run flamegraph
Push-Location rust
$startTime = Get-Date
try {
    & cargo @flameArgs
    $exitCode = $LASTEXITCODE
} finally {
    Pop-Location
}
$endTime = Get-Date
$elapsed = $endTime - $startTime

Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

if ($exitCode -eq 0 -and (Test-Path $outputFile)) {
    Write-Host "[Done] Flamegraph generated successfully" -ForegroundColor Green
    Write-Host "[Output] $outputFile" -ForegroundColor Green
    Write-Host "[Duration] $($elapsed.ToString('hh\:mm\:ss\.fff'))" -ForegroundColor DarkGray

    if ($Open) {
        Write-Host "[Opening] Launching in browser..." -ForegroundColor Cyan
        Start-Process $outputFile
    }
} else {
    Write-Host "[Error] Flamegraph generation failed with exit code $exitCode" -ForegroundColor Red
    exit 1
}

Write-Host ""
```

Create scripts/profile/run_pyspy.ps1 for combined Python+Rust profiling:

```powershell
<#
.SYNOPSIS
    Profiles Python+Rust combined stacks using py-spy.

.DESCRIPTION
    Runs py-spy with configurable modes for profiling CLASSIC's hybrid Python-Rust execution.
    Uses --native flag to capture Rust frames from PyO3 extensions.

    Modes:
    - quick (default): 99 Hz sampling, 10s duration
    - thorough: 997 Hz sampling, 60s duration

.PARAMETER Mode
    Profiling mode: 'quick' (default) or 'thorough'.

.PARAMETER EntryPoint
    Python entry point to profile. Default: CLASSIC_Interface.py

.PARAMETER Native
    Include native (Rust) frames. Default: enabled.

.PARAMETER NoNative
    Disable native frame capture (Python-only profile).

.PARAMETER Output
    Custom output path. Default: target/profiling/py-spy/profile-{timestamp}.svg

.PARAMETER Format
    Output format: 'flamegraph' (default), 'speedscope', 'raw'.

.PARAMETER Duration
    Custom duration in seconds (overrides mode default).

.PARAMETER Open
    Open the generated output in default application.

.EXAMPLE
    .\run_pyspy.ps1
    # Quick profile of GUI with native frames

.EXAMPLE
    .\run_pyspy.ps1 -Mode thorough -EntryPoint CLASSIC_ScanLogs.py
    # Thorough profile of CLI

.EXAMPLE
    .\run_pyspy.ps1 -NoNative -Format speedscope
    # Python-only profile in speedscope format
#>

[CmdletBinding()]
param(
    [ValidateSet('quick', 'thorough')]
    [string]$Mode = 'quick',

    [string]$EntryPoint = 'CLASSIC_Interface.py',

    [switch]$Native = $true,

    [switch]$NoNative,

    [string]$Output,

    [ValidateSet('flamegraph', 'speedscope', 'raw')]
    [string]$Format = 'flamegraph',

    [int]$Duration = 0,

    [switch]$Open
)

$env:PROFILE_MODE = $Mode

Write-Host "========================================" -ForegroundColor Cyan
Write-Host " CLASSIC py-spy Profiler" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Determine parameters based on mode
$rate = if ($Mode -eq 'thorough') { 997 } else { 99 }
$dur = if ($Duration -gt 0) { $Duration } elseif ($Mode -eq 'thorough') { 60 } else { 10 }

Write-Host "[Config] Mode: $Mode" -ForegroundColor Yellow
Write-Host "         Entry point: $EntryPoint" -ForegroundColor DarkGray
Write-Host "         Sampling rate: $rate Hz" -ForegroundColor DarkGray
Write-Host "         Duration: ${dur}s" -ForegroundColor DarkGray

# Handle native flag
$useNative = $Native -and -not $NoNative
Write-Host "         Native frames: $(if ($useNative) { 'enabled' } else { 'disabled' })" -ForegroundColor DarkGray

# Setup output
$timestamp = Get-Date -Format "yyyy-MM-dd-HHmmss"
$outputDir = "target/profiling/py-spy"
New-Item -ItemType Directory -Force -Path $outputDir | Out-Null

$ext = switch ($Format) {
    'speedscope' { 'json' }
    'raw' { 'txt' }
    default { 'svg' }
}

$outputFile = if ($Output) {
    $Output
} else {
    "$outputDir/profile-$timestamp.$ext"
}

Write-Host "[Output] $outputFile" -ForegroundColor Yellow
Write-Host "[Format] $Format" -ForegroundColor Yellow

# Build py-spy command
$pyspyArgs = @(
    "record",
    "-o", $outputFile,
    "-r", $rate,
    "-d", $dur,
    "-f", $Format
)

if ($useNative) {
    $pyspyArgs += "--native"
}

# Add Python command
$pyspyArgs += @("--", "uv", "run", "python", $EntryPoint)

Write-Host ""
Write-Host "[Command] py-spy $($pyspyArgs -join ' ')" -ForegroundColor DarkGray
Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host "NOTE: py-spy requires elevated privileges on Windows." -ForegroundColor Yellow
Write-Host "If profiling fails, run PowerShell as Administrator." -ForegroundColor Yellow
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

# Check if py-spy is installed
$pyspyPath = Get-Command py-spy -ErrorAction SilentlyContinue
if (-not $pyspyPath) {
    Write-Host "[Error] py-spy not found. Install with: pip install py-spy" -ForegroundColor Red
    exit 1
}

# Run py-spy
$startTime = Get-Date
try {
    & py-spy @pyspyArgs
    $exitCode = $LASTEXITCODE
} catch {
    Write-Host "[Error] $_" -ForegroundColor Red
    $exitCode = 1
}
$endTime = Get-Date
$elapsed = $endTime - $startTime

Write-Host ""
Write-Host "----------------------------------------" -ForegroundColor Cyan
Write-Host ""

if ($exitCode -eq 0 -and (Test-Path $outputFile)) {
    Write-Host "[Done] Profile generated successfully" -ForegroundColor Green
    Write-Host "[Output] $outputFile" -ForegroundColor Green
    Write-Host "[Duration] $($elapsed.ToString('hh\:mm\:ss\.fff'))" -ForegroundColor DarkGray

    if ($Open) {
        Write-Host "[Opening] Launching output..." -ForegroundColor Cyan
        Start-Process $outputFile
    }
} else {
    Write-Host "[Error] py-spy profiling failed with exit code $exitCode" -ForegroundColor Red
    Write-Host ""
    Write-Host "Common issues:" -ForegroundColor Yellow
    Write-Host "  - Need Administrator privileges (Run as Administrator)" -ForegroundColor DarkGray
    Write-Host "  - Application exited before profiling completed" -ForegroundColor DarkGray
    Write-Host "  - Entry point not found: $EntryPoint" -ForegroundColor DarkGray
    exit 1
}

Write-Host ""
```
