   ```toml
   [package]
   name = "classic-gui"
   version = "9.0.0"
   edition = "2024"
   rust-version = "1.85.0"
   authors = ["CLASSIC Development Team"]
   description = "Rust-native GUI for CLASSIC crash log analyzer"
   repository = "https://github.com/evildarkarchon/CLASSIC-Fallout4"

   [[bin]]
   name = "classic-gui"
   path = "src/main.rs"

   [lib]
   crate-type = ["rlib"]

   [dependencies]
   slint = { workspace = true, features = ["backend-winit", "renderer-skia", "image-default-formats"] }
   classic-shared-core = { path = "../../foundation/classic-shared-core", features = ["gui-bridge"] }
   tokio-util = "0.7"

   [build-dependencies]
   slint-build = "1.15"

   [lints.rust]
   deprecated = "deny"
   rust_2024_compatibility = "deny"
   unsafe_code = "deny"
   missing_docs = "warn"
   unused = "deny"
   ```

5. Create rust/ui-applications/classic-gui/build.rs:
   ```rust
   //! Build script for classic-gui - compiles Slint UI files to Rust

   fn main() {
       slint_build::compile("ui/main.slint").expect("Slint compilation failed");
   }
   ```

6. Create rust/ui-applications/classic-gui/src/lib.rs:
   ```rust
   //! CLASSIC GUI - Rust-native graphical interface
   //!
   //! This crate provides the Slint-based GUI for CLASSIC crash log analyzer.
   //! It uses the existing Tokio runtime from classic-shared-core and the
   //! AsyncBridge for coordinating between UI and async operations.

   // Re-export for potential testing
   pub use classic_shared_core::AsyncBridge;
   ```

7. Create rust/ui-applications/classic-gui/src/main.rs:
   ```rust
   //! CLASSIC GUI - Application entry point
   //!
   //! Initializes the Slint GUI application and runs the event loop.
   //! Uses the global Tokio runtime from classic-shared-core (ONE RUNTIME RULE).

   slint::include_modules!();

   use classic_shared_core::get_runtime;

   fn main() {
       // Initialize global Tokio runtime before Slint event loop
       // This ensures ONE RUNTIME RULE compliance
       let _ = get_runtime();

       // Create main window
       let window = MainWindow::new().expect("Failed to create main window");

       // Run Slint event loop (blocks until window closes)
       window.run().expect("Failed to run application");
   }
   ```

IMPORTANT: Do NOT use #[tokio::main] - that would create a second runtime violating ONE RUNTIME RULE.

```slint
// CLASSIC GUI - Main Window
// Scaffold UI matching Python GUI layout

import { Button, ProgressIndicator, TabWidget, VerticalBox, HorizontalBox, LineEdit, CheckBox, GroupBox } from "std-widgets.slint";

export component MainWindow inherits Window {
    // Window properties - match Python GUI dimensions
    title: "Crash Log Auto Scanner & Setup Integrity Checker | CLASSIC v9.0.0";
    icon: @image-url("../assets/CLASSIC.ico");
    preferred-width: 650px;
    preferred-height: 580px;
    min-width: 550px;
    min-height: 580px;

    // Properties accessible from Rust
    in-out property <float> progress: 0;
    in-out property <string> status: "Ready";
    in-out property <bool> scan-in-progress: false;

    // Callbacks invoked from UI (will be wired in Plan 02)
    callback start-scan();
    callback cancel-scan();

    VerticalBox {
        padding: 10px;
        spacing: 10px;

        TabWidget {
            Tab {
                title: "MAIN OPTIONS";
                VerticalBox {
                    padding: 10px;
                    spacing: 8px;

                    // Game selection placeholder
                    GroupBox {
                        title: "Game Selection";
                        HorizontalBox {
                            spacing: 10px;
                            Text { text: "Game:"; vertical-alignment: center; }
                            Text { text: "Fallout 4 (placeholder)"; vertical-alignment: center; }
                        }
                    }

                    // Scan options placeholder
                    GroupBox {
                        title: "Scan Options";
                        VerticalBox {
                            spacing: 5px;
                            CheckBox { text: "Scan crash logs"; checked: true; }
                            CheckBox { text: "Scan game files"; }
                            CheckBox { text: "Generate statistics"; }
                        }
                    }

                    // Spacer
                    Rectangle { }

                    // Action buttons
                    HorizontalBox {
                        spacing: 10px;
                        Button {
                            text: scan-in-progress ? "Scanning..." : "Scan Crash Logs";
                            enabled: !scan-in-progress;
                            clicked => { root.start-scan(); }
                        }
                        Button {
                            text: "Cancel";
                            enabled: scan-in-progress;
                            clicked => { root.cancel-scan(); }
                        }
                    }
                }
            }

            Tab {
                title: "FILE BACKUP";
                VerticalBox {
                    padding: 10px;
                    Text { text: "Backup options will be implemented in Phase 24"; }
                }
            }

            Tab {
                title: "ARTICLES";
                VerticalBox {
                    padding: 10px;
                    Text { text: "Help articles will be implemented in later phases"; }
                }
            }

            Tab {
                title: "RESULTS";
                VerticalBox {
                    padding: 10px;
                    Text { text: "Scan results will appear here after Phase 22"; }
                }
            }
        }

        // Progress area - always visible at bottom
        VerticalBox {
            spacing: 5px;
            ProgressIndicator {
                progress: root.progress / 100;
            }
            Text {
                text: root.status;
                horizontal-alignment: center;
            }
        }
    }
}
```
